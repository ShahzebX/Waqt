01 · Stage 1

App Concept & Features

Waqt (وقت — "time" in Urdu) is a daily planner that fetches your prayer times automatically, then helps you build study sessions, tasks, and breaks in the gaps between prayers. The prayers are fixed anchors — everything else flows around them.

Why "Waqt" impresses your teacher: It covers an API network call, device sensor (compass), local database (Room), background scheduling (WorkManager), navigation between 4+ screens, and live UI updates — all in one clean app. That's a comprehensive project, not a toy.

Aladhan API Integration

Aladhan API — free, no API key, no registration. Base URL: https://api.aladhan.com/v1/. Returns Fajr, Dhuhr, Asr, Maghrib, Isha times for any lat/long or city name.

AladhanApi.kt — Retrofit interface

network/AladhanApi.kt
interface AladhanApi {
    @GET("timings")
    suspend fun getPrayerTimes(
        @Query("latitude")  latitude:  Double,
        @Query("longitude") longitude: Double,
        @Query("method")    method:    Int = 18 // Karachi method
    ): PrayerResponse

    @GET("timingsByCity")
    suspend fun getPrayerTimesByCity(
        @Query("city")    city:    String,
        @Query("country") country: String = "PK",
        @Query("method")  method:  Int = 18
    ): PrayerResponse
}

// Retrofit instance (singleton)
object RetrofitInstance {
    val api: AladhanApi by lazy {
        Retrofit.Builder()
            .baseUrl("https://api.aladhan.com/v1/")
            .addConverterFactory(GsonConverterFactory.create())
            .build()
            .create(AladhanApi::class.java)
    }
}

PrayerRepository.kt — API call with Room caching

repository/PrayerRepository.kt
class PrayerRepository(
    private val api: AladhanApi,
    private val dao: PrayerDao
) {
    suspend fun getPrayerTimes(lat: Double, lng: Double): Result<List<Prayer>> {
        return try {
            // 1. Try network first
            val response = api.getPrayerTimes(lat, lng)
            val prayers = response.data.timings.toEntityList()
            // 2. Cache in Room
            dao.insertAll(prayers)
            Result.success(prayers.map { it.toDomain() })
        } catch (e: Exception) {
            // 3. Offline fallback — return cached data
            val cached = dao.getTodayPrayers()
            if (cached.isNotEmpty())
                Result.success(cached.map { it.toDomain() })
            else
                Result.failure(e)
        }
    }
}

@Composable
fun QiblaScreen() {
    val vm: QiblaViewModel = viewModel()
    val azimuth by vm.azimuth.collectAsState(0f)
    val qiblaAngle by vm.qiblaAngle.collectAsState(0f)

    Box(contentAlignment = Alignment.Center, modifier = Modifier.fillMaxSize()) {
        Canvas(modifier = Modifier.size(280.dp)) {
            val center = Offset(size.width/2, size.height/2)
            val radius = size.minDimension / 2

            // Draw compass circle
            drawCircle(color = compassBg, radius = radius)
            drawCircle(color = compassBorder, radius = radius,
                       style = Stroke(width = 3.dp.toPx()))

            // Rotate needle to point at Qibla
            val angle = qiblaAngle - azimuth
            rotate(angle, center) {
                drawLine(
                    color = Color.Green,
                    start = center,
                    end = Offset(center.x, center.y - radius * 0.8f),
                    strokeWidth = 6.dp.toPx(),
                    cap = StrokeCap.Round
                )
            }
        }
        Text("🕋", fontSize = 32.sp,
             modifier = Modifier.align(Alignment.TopCenter).padding(top = 24.dp))
    }
}

07 · Stage 1

Study Planner Logic

The planner's key feature is auto-generating study blocks in the gaps between prayers. This is the algorithm your teacher will find impressive.

viewmodel/PlannerViewModel.kt — slot generation
fun generateStudySlots(prayers: List<Prayer>): List<TimeSlot> {
    val slots = mutableListOf<TimeSlot>()
    val sorted = prayers.sortedBy { it.epochMs }

    for (i in 0 until sorted.size - 1) {
        val gapMinutes = (sorted[i+1].epochMs - sorted[i].epochMs) / 60_000

        when {
            gapMinutes < 45  -> /* too short — skip */
                Unit
            gapMinutes < 90  -> slots.add(TimeSlot(
                start    = sorted[i].time,
                end      = sorted[i+1].time,
                type     = SlotType.SHORT_STUDY,
                label    = "Quick revision (${gapMinutes}m)"
            ))
            gapMinutes < 180 -> slots.add(TimeSlot(
                start    = sorted[i].time,
                end      = sorted[i+1].time,
                type     = SlotType.STUDY,
                label    = "Study block (${gapMinutes}m)"
            ))
            else              -> {
                // Long gap — split into study + break
                slots.add(TimeSlot(type = SlotType.STUDY, label = "Deep work"))
                slots.add(TimeSlot(type = SlotType.BREAK, label = "Break"))
            }
        }
    }
    return slots
}

Prayer notification worker

worker/PrayerNotificationWorker.kt
class PrayerNotificationWorker(ctx: Context, params: WorkerParameters)
    : CoroutineWorker(ctx, params) {

    override suspend fun doWork(): Result {
        val prayerName = inputData.getString("prayer_name") ?: return Result.failure()
        val prayerTime = inputData.getString("prayer_time") ?: return Result.failure()

        showNotification(
            title = "$prayerName in 10 minutes",
            body  = "Time: $prayerTime — wrap up your current task"
        )
        return Result.success()
    }
}

// Schedule all 5 prayers after fetching times
fun schedulePrayerNotifications(prayers: List<Prayer>) {
    prayers.forEach { prayer ->
        val delayMs = prayer.epochMs - System.currentTimeMillis() - (10 * 60_000)
        if (delayMs > 0) {
            val request = OneTimeWorkRequestBuilder<PrayerNotificationWorker>()
                .setInitialDelay(delayMs, TimeUnit.MILLISECONDS)
                .setInputData(workDataOf(
                    "prayer_name" to prayer.name,
                    "prayer_time" to prayer.time
                )).build()
            WorkManager.getInstance(context).enqueue(request)
        }
    }
}
